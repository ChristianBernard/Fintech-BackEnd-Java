-- Creation SQL Commands
CREATE TABLE Conta (
    ID_Conta INTEGER PRIMARY KEY,
    Tipo_Conta VARCHAR2(30),
    Saldo DECIMAL(15,2),
    Status VARCHAR2(20),
    Data_Abertura DATE,
    Usuario_ID_Usuario INTEGER
);

CREATE TABLE Pagamento (
    ID_Pagamento INTEGER PRIMARY KEY,
    ID_Conta INTEGER,
    Tipo_Pagamento VARCHAR2(30),
    Valor DECIMAL(15,2),
    Data_Pagamento DATE,
    Descricao VARCHAR2(200),
    Status VARCHAR2(20),
    FOREIGN KEY (ID_Conta) REFERENCES Conta(ID_Conta)
);
